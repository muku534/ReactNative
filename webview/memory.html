Memory Monitor

Track your React Native app's memory usage in real-time. Add this snippet to your app's entry file (e.g., App.js or index.js).

if (__DEV__) { const memWs = new WebSocket('ws://localhost:8348'); // Listen for Force GC requests from VS Code memWs.onmessage = (e) => { try { const msg = JSON.parse(e.data); if (msg.type === 'forceGC' && global.gc) global.gc(); } catch(err) {} }; setInterval(() => { if (memWs.readyState === WebSocket.OPEN && global.HermesInternal) { const stats = global.HermesInternal.getInstrumentedStats(); memWs.send(JSON.stringify({ type: 'memory', timestamp: Date.now(), heapUsed: stats.js_heapSize || 0, heapTotal: stats.js_totalHeapSize || stats.js_heapSize || 0, gcCount: stats.js_numGCs || 0, gcTime: stats.js_gcTime || 0, mallocSize: stats.js_mallocSize || 0, objectCount: stats.js_numLiveObjects || 0, externalBytes: stats.js_externalBytes || 0 })); } }, 1000); }
